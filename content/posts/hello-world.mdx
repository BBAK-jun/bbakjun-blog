### JavaScript/TypeScript

```typescript
interface User {
  id: number;
  name: string;
  email: string;
  isActive: boolean;
}

class UserService {
  private users: User[] = [];

  async createUser(userData: Omit<User, 'id'>): Promise<User> {
    const newUser: User = {
      id: Math.random(),
      ...userData
    };

    this.users.push(newUser);
    console.log(`사용자 생성됨: ${newUser.name}`)

    return newUser;
  }

  getUsersByStatus(active: boolean): User[] {
    return this.users.filter(user => user.isActive === active);
  }
}

// 사용법
const userService = new UserService();
const user = await userService.createUser({
  name: "빡준",
  email: "bbakjun@example.com",
  isActive: true
});
```

### Python

```python
from typing import List, Optional
import asyncio

class BlogPost:
    def __init__(self, title: str, content: str, tags: List[str]):
        self.title = title
        self.content = content
        self.tags = tags
        self.views = 0

    async def increment_views(self) -> int:
        """조회수를 비동기적으로 증가시킵니다."""
        await asyncio.sleep(0.1)  # Redis 호출 시뮬레이션
        self.views += 1
        return self.views

    def __str__(self) -> str:
        return f"📝 {self.title} (조회수: {self.views})"

# 블로그 포스트 생성
post = BlogPost(
    title="Hello World",
    content="첫 번째 포스트입니다!",
    tags=["nextjs", "python", "tutorial"]
)

print(post)  # 📝 Hello World (조회수: 0)
```

### Bash/Shell

```bash
#!/bin/bash

# 블로그 배포 스크립트
deploy_blog() {
    echo "🚀 블로그 배포를 시작합니다..."

    # 의존성 설치
    npm install

    # 빌드
    npm run build

    # 테스트
    if npm test; then
        echo "✅ 테스트 통과!"
    else
        echo "❌ 테스트 실패"
        exit 1
    fi

    # Vercel 배포
    vercel --prod

    echo "🎉 배포 완료!"
}

deploy_blog
```
